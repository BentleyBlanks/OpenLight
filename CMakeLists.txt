CMAKE_MINIMUM_REQUIRED ( VERSION 2.8.4 )
project(OpenLight)

IF(WIN32)
    SET(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} /SUBSYSTEM:WINDOWS") 
ENDIF(WIN32)

SET(OPENLIGHT_SRC Source)
SET(OPENLIGHT_SHADER Shader)

set_property(GLOBAL PROPERTY USE_FOLDERS ON)

include_directories(${OPENLIGHT_SRC}/core)

FILE(GLOB SHADER_SOURCE ${OPENLIGHT_SHADER}/*.hlsl)
SOURCE_GROUP("Shader" FILES ${SHADER_SOURCE})


ADD_SUBDIRECTORY(${OPENLIGHT_SRC})
